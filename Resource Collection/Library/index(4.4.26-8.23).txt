        async function processLogin() {
            const userid = document.getElementById('userid').value.trim();
            const pin = document.getElementById('pin').value.trim();
            const errorDiv = document.getElementById('loginError');
            
            if(!userid || !pin) { errorDiv.innerText = "Isi User ID dan PIN."; errorDiv.classList.remove('hidden'); return; }

            errorDiv.classList.add('hidden');
            document.getElementById('btnLoginText').innerText = "Memeriksa Akun..."; 
            document.getElementById('loginLoader').classList.remove('hidden');
            document.getElementById('btnLogin').disabled = true;

            try {
                // Di backend, kita hanya mengirimkan userid & pin. Backend akan mengenali role
                const response = await fetch(SCRIPT_URL, {
                    method: 'POST', headers: { 'Content-Type': 'text/plain;charset=utf-8' }, 
                    body: JSON.stringify({ action: 'login', userid: userid, pin: pin })
                });

                const result = await response.json();

                if (result.status === 'success') {
                    currentUserData = result.data;
                    globalCampaigns = result.campaigns || [];
                    globalCustomers = result.customers || [];
                    globalDocuments = result.documents || []; 
                    
                    document.getElementById('loginSection').classList.add('hidden');
                    
                    // APPLY RBAC LOGIC ON UI
                    applyRoleBasedUI(currentUserData);

                    document.getElementById('dashboardSection').classList.remove('hidden');
                    document.getElementById('userInfo').classList.remove('hidden');
                    
                    document.getElementById('displayUser').innerHTML = `<i class="fas fa-user mr-1"></i> ${currentUserData.userId}`;
                    document.getElementById('displayRole').innerText = currentUserData.role; // Tampilkan role asli (bisa text manual)
                    document.getElementById('depot').value = currentUserData.depot;

                    // Logika tampilan nama identitas (Admin vs Head Office)
                    // Cari tahu apakah digolongkan secara sistem sebagai non-osc
                    let sysRole = currentUserData.role.toLowerCase();
                    if (sysRole !== 'admin' && sysRole !== 'osc' && sysRole !== 'supervisor') {
                        sysRole = 'non-osc';
                    }

                    if (sysRole === 'admin') {
                        document.getElementById('displayDepot').innerText = `${currentUserData.depot} - ${currentUserData.kodeDepot || '000'}`;
                    } else {
                        document.getElementById('displayDepot').innerText = `Head Office - 0`;
                    }

                    logActivity("User Login Berhasil");

                } else {
                    errorDiv.innerText = result.message || "Kredensial salah."; errorDiv.classList.remove('hidden');
                }
            } catch (error) {
                errorDiv.innerText = "Terjadi kesalahan jaringan."; errorDiv.classList.remove('hidden');
            } finally {
                document.getElementById('btnLoginText').innerText = "Masuk ke Portal";
                document.getElementById('loginLoader').classList.add('hidden');
                document.getElementById('btnLogin').disabled = false;
            }
        }

        function handleNoDocInput() {
            let val = document.getElementById('noDoc').value.trim();
            let statusSpan = document.getElementById('docStatus');
            
            if(!val) {
                statusSpan.innerHTML = '<span class="text-blue-600">(Auto Fill)</span>';
                return;
            }

            let foundDoc = globalDocuments.find(d => d.noDoc && d.noDoc.trim().toUpperCase() === val.toUpperCase());
            if(foundDoc) {
                statusSpan.innerHTML = '<span class="text-green-600"><i class="fas fa-check-circle"></i> Terdaftar di Source DB</span>';
                document.getElementById('custCode').value = foundDoc.custCode;
                document.getElementById('netValue').value = formatCurrencyValue(foundDoc.netValue);
                document.getElementById('vols').value = foundDoc.vols;
                document.getElementById('cashback').value = formatCurrencyValue(foundDoc.cashback);
                searchCustomer(foundDoc.custCode, foundDoc.custName);
            } else {
                if(val.length > 8) {
                    statusSpan.innerHTML = '<span class="text-orange-500"><i class="fas fa-info-circle"></i> Tidak ada di Source DB</span>';
                }
                document.getElementById('custCode').value = ""; document.getElementById('netValue').value = "";
                document.getElementById('vols').value = ""; document.getElementById('cashback').value = "";
                document.getElementById('custName').value = "";
            }

            let parts = val.split('/');
            let codePart = parts.find(p => p.length === 4 && /^\d+$/.test(p));
            if(codePart) {
                let fullCode = codePart.substring(0, 2) + "20" + codePart.substring(2, 4);   
                document.getElementById('codeCampaign').value = fullCode;
                searchCampaign(fullCode); 
            } else {
                document.getElementById('codeCampaign').value = ""; searchCampaign("");
            }
        }

                let data = new Uint8Array(e.target.result);
                let workbook = XLSX.read(data, {type: 'array'});
                let firstSheet = workbook.Sheets[workbook.SheetNames[0]];
                
                let rows = XLSX.utils.sheet_to_json(firstSheet, {header: 1, raw: false, defval: ""});
                
                if(rows.length < 3) {
                    showNotification("File Excel Kosong atau format salah (Minimal baris ke-3)!", "error");
                    return;
                }

                // Ambil header dari baris ke-2 (index 1)
                let headers = rows[1].map(h => String(h).toLowerCase().trim());
                let idxDoc = -1, idxKtp = -1, idxNpwp = -1, idxLunas = -1, idxBln = -1, idxDay = -1;
                
                headers.forEach((h, i) => {
                    if(h.includes("no doc")) idxDoc = i;
                    if(h.includes("ktp")) idxKtp = i;
                    if(h.includes("npwp")) idxNpwp = i;
                    if(h.includes("lunas")) idxLunas = i;
                    if(h.includes("bulan") || h.includes("tagihan")) idxBln = i;
                    if(h.includes("top") || h.includes("day") || h.includes("lewat")) idxDay = i;
                });

                if(idxDoc === -1) idxDoc = 0; if(idxKtp === -1) idxKtp = 1;
                if(idxNpwp === -1) idxNpwp = 2; if(idxLunas === -1) idxLunas = 3;
                if(idxBln === -1) idxBln = 4; if(idxDay === -1) idxDay = 5;

                massUploadDataValid = [];
                let localUploadedDocs = []; 
                let tbody = document.getElementById('massTableBody');
                tbody.innerHTML = '';
                
                let countValid = 0;
                let countNotInDB = 0;

                // Loop data dimulai dari baris ke-3 (index 2)
                for(let i=2; i<rows.length; i++) {
                    let row = rows[i];
                    let noDocRaw = String(row[idxDoc] || "").trim(); 
                    if(!noDocRaw) continue; 

                    let noDocUpper = noDocRaw.toUpperCase();
                    let isDuplicateInFile = localUploadedDocs.includes(noDocUpper);
                    let foundDb = globalDocuments.find(d => d.noDoc && d.noDoc.trim().toUpperCase() === noDocUpper);
                    
                    let statusHtml = "";
                    let rowClass = "";

                    if(isDuplicateInFile) {
                        statusHtml = '<span class="px-2 py-1 bg-red-100 text-red-700 rounded text-[10px] font-bold">Duplikat di Excel</span>';
                        rowClass = "bg-red-50 opacity-60";
                    } else if (!foundDb) {
                        statusHtml = '<span class="px-2 py-1 bg-orange-100 text-orange-700 rounded text-[10px] font-bold">Tdk di Source DB</span>';
                        rowClass = "bg-orange-50 opacity-60";
                        countNotInDB++;
                    } else {
                        statusHtml = '<span class="px-2 py-1 bg-green-100 text-green-700 rounded text-[10px] font-bold">Siap di-Cek Server</span>';
                        rowClass = "hover:bg-gray-50";
                        countValid++;
                        localUploadedDocs.push(noDocUpper);

                        let manualKtp = String(row[idxKtp]||"").trim();
                        let manualNpwp = String(row[idxNpwp]||"").trim();
                        let manualTglLunas = String(row[idxLunas]||"").trim(); 
                        let manualDay = String(row[idxDay]||"").trim();
                        let manualBlnTagihan = String(row[idxBln]||"").trim();

                        massUploadDataValid.push({
                            tglClaim: defaultDateStr,
                            noDoc: noDocRaw,
                            codeCampaign: "",
                            campaign: "",
                            custCode: "",
                            custName: "",
                            netValue: "",
                            vols: "",
                            cashback: "",
                            periodStart: "",
                            campaignEnd: "",
                            noKtp: manualKtp,
                            noNpwp: manualNpwp,
                            statusPajak: "",
                            tglLunas: manualTglLunas,
                            lewatTop: manualDay,
                            bulanTagihan: manualBlnTagihan
                        });
                    }

                    let identitas = String(row[idxKtp]||"") ? String(row[idxKtp]) : (String(row[idxNpwp]||"") ? String(row[idxNpwp]) : "-");
                    
                    tbody.innerHTML += `
                        <tr class="${rowClass}">
                            <td class="px-4 py-2 border-b">${statusHtml}</td>
                            <td class="px-4 py-2 border-b font-medium text-blue-800">${noDocRaw}</td>
                            <td class="px-4 py-2 border-b text-gray-600">${identitas}</td>
                            <td class="px-4 py-2 border-b font-medium">${String(row[idxLunas]||"-").trim()}</td>
                            <td class="px-4 py-2 border-b font-medium">${String(row[idxBln]||"-").trim()}</td>
                            <td class="px-4 py-2 border-b text-red-600 font-bold">${String(row[idxDay]||"-").trim()}</td>
                        </tr>
                    `;
                }

                document.getElementById('massCount').innerText = `${countValid} Siap Proses, ${countNotInDB} Tidak dikenal`;
                document.getElementById('massPreviewContainer').classList.remove('hidden');
                
                let btnMass = document.getElementById('btnSubmitMass');
                if(countValid > 0 && currentUserData.aksesMass === 'yes') {
                    btnMass.disabled = false;
                    btnMass.classList.remove('bg-gray-400', 'cursor-not-allowed');
                } else {
                    btnMass.disabled = true;
                    btnMass.classList.add('bg-gray-400', 'cursor-not-allowed');
                    if (currentUserData.aksesMass !== 'yes') {
                         showNotification("Anda tidak memiliki hak akses Mass Upload.", "warning");
                    } else {
                         showNotification("Tidak ada data valid yang bisa dikirim ke server.", "warning");
                    }
                }
            };
            reader.readAsArrayBuffer(file);
        });

            if (sysRole !== 'admin' && sysRole !== 'osc' && sysRole !== 'supervisor') {
                sysRole = 'non-osc';
            }
            
            // Logic restriction based on current user role
            if (sysRole === 'supervisor') {
                spvOpts.forEach(o => o.classList.remove('hidden'));
                roleSelect.disabled = false;
                roleSelect.classList.remove('bg-gray-100');
            } else {

                spvOpts.forEach(o => o.classList.add('hidden'));
                roleSelect.disabled = (u.role !== 'admin');
                if(roleSelect.disabled) roleSelect.classList.add('bg-gray-100');
            }
            roleSelect.value = u.role || 'admin';

            document.getElementById('editAccessModal').classList.remove('hidden');
        }

        async function loadAddAccountForm() {

                sysRole = 'non-osc';
            }

            roleSelect.innerHTML = '';
            depotSelect.innerHTML = '<option value="" disabled selected>Memuat Depot...</option>';

            // Setup Options based on Hierarchy
            if (sysRole === 'supervisor') {
                roleSelect.innerHTML += '<option value="admin">Admin</option>';
                roleSelect.innerHTML += '<option value="osc">OSC</option>';
                roleSelect.innerHTML += '<option value="non-osc">Non-OSC</option>';

                roleSelect.innerHTML += '<option value="admin">Admin</option>';
                roleSelect.innerHTML += '<option value="osc_req">Request Akun OSC</option>';
                roleSelect.innerHTML += '<option value="non-osc_req">Request Akun Non-OSC</option>';

                document.getElementById('btnAddAccountText').innerText = "Simpan Data / Request";
            } else if (sysRole === 'admin') {
                roleSelect.innerHTML += '<option value="admin_req">Request Akun Admin</option>';
                
                document.getElementById('addAccountTitle').innerHTML = '<i class="fas fa-user-plus text-green-600 mr-2"></i>Request Akun Admin';
                document.getElementById('btnAddAccountText').innerText = "Kirim Permintaan (Request)";
            }

            // Setup Depot Options
            if (sysRole === 'admin') {
                // Admin locked to their own depot
                depotSelect.innerHTML = `<option value="${currentUserData.depot}" selected>${currentUserData.depot}</option>`;
                depotSelect.disabled = true;
                handleAddRoleChange(); // Trigger UI logic
            } else {
                try {
                    const response = await fetch(SCRIPT_URL, {
                        method: 'POST', headers: { 'Content-Type': 'text/plain;charset=utf-8' },
                        body: JSON.stringify({ action: 'getManageableData', requesterId: currentUserData.userId, requesterRole: currentUserData.role })
                    });
                    const result = await response.json();
                    
                    if (result.status === 'success') {
                        currentManageableUsers = result.users || [];
                        const opts = (sysRole === 'supervisor') ? result.allDepots : result.managedDepots;
                        globalAvailableDepots = [...new Set(opts)].filter(d => d.trim() !== '');
                        
                        // Populate OSC Dropdown
                        const oscSelect = document.getElementById('addAssignOsc');
                        oscSelect.innerHTML = '<option value="" disabled selected>-- Pilih OSC --</option>';
                        const oscUsers = currentManageableUsers.filter(u => u.role.toLowerCase() === 'osc');
                        if (oscUsers.length > 0) {
                            oscUsers.forEach(u => {
                                oscSelect.innerHTML += `<option value="${u.userId}">${u.userId} (${u.depotName})</option>`;
                            });
                        } else {
                            oscSelect.innerHTML = '<option value="" disabled selected>Tidak ada OSC tersedia</option>';
                        }

                        handleAddRoleChange(); // Trigger UI logic only after depots are loaded
                    }
                } catch(e) {
                    depotSelect.innerHTML = '<option value="" disabled selected>Gagal memuat depot</option>';
                }
            }
        }

        function handleAddRoleChange() {
            const roleSelected = document.getElementById('addRole').value;
            
            let sysRole = currentUserData.role.toLowerCase();
            if (sysRole !== 'admin' && sysRole !== 'osc' && sysRole !== 'supervisor') {
                sysRole = 'non-osc';
            }
            
            const deptContainer = document.getElementById('addDeptContainer');
            const oscContainer = document.getElementById('assignOscContainer');
            const assignDepotsContainer = document.getElementById('assignDepotsContainer');
            const depotSelect = document.getElementById('addDepotName');
            const assignDepotsList = document.getElementById('assignDepotsList');

            // Reset visibilitas
            deptContainer.classList.add('hidden');
            oscContainer.classList.add('hidden');
            assignDepotsContainer.classList.add('hidden');
            
            const isOscOrNonOsc = roleSelected.includes('osc') || roleSelected.includes('non-osc');
            const isAdmin = roleSelected.includes('admin');

            // --- Atur Penempatan Depot Utama ---
            if (isOscOrNonOsc) {
                depotSelect.innerHTML = `<option value="Head Office" selected>Head Office</option>`;
                depotSelect.disabled = true;
            } else {
                // Kalau Admin, kembalikan daftar depot
                if (sysRole === 'admin') {
                    depotSelect.innerHTML = `<option value="${currentUserData.depot}" selected>${currentUserData.depot}</option>`;
                    depotSelect.disabled = true;
                } else {
                    depotSelect.disabled = false;
                    depotSelect.innerHTML = '<option value="" disabled selected>-- Pilih Depot --</option>';
                    globalAvailableDepots.forEach(d => {
                        depotSelect.innerHTML += `<option value="${d}">${d}</option>`;
                    });
                }
            }

            // --- Atur Multi-Select Assign Depot ---
            if (isOscOrNonOsc) {
                assignDepotsContainer.classList.remove('hidden');
                assignDepotsList.innerHTML = '';
                
                // Tambahkan fitur All Depot HANYA untuk Non-OSC
                if (roleSelected.includes('non-osc')) {
                    assignDepotsList.innerHTML += `
                        <label class="flex items-center space-x-2 text-sm text-gray-700 p-2 hover:bg-purple-100 rounded cursor-pointer border-b border-purple-200 mb-1">
                            <input type="checkbox" value="ALL_DEPOT" onchange="toggleAllDepots(this)" class="rounded text-purple-600 focus:ring-purple-500">
                            <span class="font-bold text-purple-800">Pilih Semua Depot (All Depot)</span>
                        </label>
                    `;
                }

                globalAvailableDepots.forEach((d, i) => {
                    assignDepotsList.innerHTML += `
                        <label class="flex items-center space-x-2 text-sm text-gray-700 p-1 hover:bg-purple-50 rounded cursor-pointer">
                            <input type="checkbox" value="${d}" class="depot-cb single-depot-cb rounded text-purple-600 focus:ring-purple-500">
                            <span>${d}</span>
                        </label>
                    `;
                });
            }

            // --- Atur Manual Input Departemen (Non-OSC) ---
            if (roleSelected.includes('non-osc')) {
                deptContainer.classList.remove('hidden');
            }

            // --- Atur Kolom Supervisor ---
            if (sysRole === 'supervisor' && isAdmin) {
                // Wajib assign ke OSC mana (hanya muncul saat bikin Admin)
                oscContainer.classList.remove('hidden');
            }

            // Ganti teks tombol berdasarkan mode (Request vs Create)
            if (roleSelected.includes('_req')) {
                document.getElementById('btnAddAccountText').innerText = "Kirim Permintaan (Request)";
            } else {
                document.getElementById('btnAddAccountText').innerText = "Buat Akun Sekarang";
            }
        }

        async function submitAddAccount() {
            const userId = document.getElementById('addUserId').value.trim();
            const pin = document.getElementById('addPin').value.trim();
            const email = document.getElementById('addEmail').value.trim();
            let selectedRole = document.getElementById('addRole').value;
            
            // Penempatan Utama
            const depotName = document.getElementById('addDepotName').value;

            // Extra Data untuk Hirarki & Assignment
            let assignOsc = "";
            let assignAdmin = "";
            let finalRole = selectedRole;

            let sysRole = currentUserData.role.toLowerCase();
            if (sysRole !== 'admin' && sysRole !== 'osc' && sysRole !== 'supervisor') {
                sysRole = 'non-osc';
            }

            // Validasi Penempatan Depot untuk Admin
            if(selectedRole.includes('admin') && !depotName) {
                showNotification("Silakan pilih Penempatan Depot terlebih dahulu.", "warning");
                return;
            }

            // Dapatkan value dari Multi-select checkbox untuk OSC/Non-OSC
            if (selectedRole.includes('osc') || selectedRole.includes('non-osc')) {
                const checkboxes = document.querySelectorAll('.single-depot-cb:checked');
                if (checkboxes.length === 0) {
                    showNotification("Silakan pilih minimal 1 Assign Akses Depot.", "warning");
                    return;
                }
                assignAdmin = Array.from(checkboxes).map(cb => cb.value).join(', ');
            }

            // Dapatkan Departemen untuk Non-OSC
            if (selectedRole.includes('non-osc')) {
                const deptDrop = document.getElementById('addDeptName').value;
                const deptMan = document.getElementById('addDeptManual').value.trim();
                
                finalRole = (deptDrop === 'Lainnya') ? deptMan : deptDrop;
                if (!finalRole) {
                    showNotification("Nama Departemen wajib diisi untuk akun Non-OSC.", "warning");
                    return;
                }
            }

            // Validasi Supervisor Extra Field
            if (sysRole === 'supervisor') {
                if (selectedRole === 'admin') {
                    assignOsc = document.getElementById('addAssignOsc').value.trim();
                    if (!assignOsc) {
                        showNotification("Kolom 'Assign ke OSC' wajib diisi saat membuat Admin.", "warning");
                        return;
                    }
                }
            }

            // Tentukan status Request
            let isRequest = false;
            if (selectedRole.includes('_req')) {
                isRequest = true;
                if (finalRole.includes('_req')) finalRole = finalRole.replace('_req', ''); 
            }

            document.getElementById('btnAddAccountText').innerText = "Memproses...";
            document.getElementById('addAccountLoader').classList.remove('hidden');
            document.getElementById('btnAddAccountSubmit').disabled = true;

            try {
                // Backend akan memproses "assignAdmin" dan "assignOsc" untuk dimasukkan ke DepoMaster
                const response = await fetch(SCRIPT_URL, {
                    method: 'POST', headers: { 'Content-Type': 'text/plain;charset=utf-8' },
                    body: JSON.stringify({ 
                        action: 'addAccount', 
                        requesterId: currentUserData.userId,
                        requesterRole: currentUserData.role,
                        data: { userId, pin, email, role: finalRole, depotName, isRequest, assignOsc, assignAdmin }
                    })
                });
                const result = await response.json();
                
                if (result.status === 'success') {
                    if (isRequest) {
                        showNotification("Request akun berhasil dikirim (Pending).", "success");
                        logActivity(`Request pembuatan akun ${userId} (${finalRole})`);
                    } else {
                        showNotification("Akun baru berhasil dibuat!", "success");
                        logActivity(`Membuat akun ${userId} (${finalRole})`);
                    }
                    document.getElementById('addAccountForm').reset();
                    handleAddRoleChange(); // reset layout
                } else {
                    // Peringatan "Depot Penuh" (Slot 1, 2, 3 full) akan dikembalikan oleh backend di result.message
                    showNotification(result.message || "Gagal memproses akun.", "error");
                }
            } catch (error) {
                showNotification("Terjadi kesalahan jaringan.", "error");
            } finally {
                document.getElementById('btnAddAccountText').innerText = document.getElementById('addRole').value.includes('_req') ? "Kirim Permintaan (Request)" : "Buat Akun Sekarang";
                document.getElementById('addAccountLoader').classList.add('hidden');
                document.getElementById('btnAddAccountSubmit').disabled = false;
            }
        }

        async function submitMassClaim() {
            if(massUploadDataValid.length === 0) return;
            
            let btnTxt = document.getElementById('btnSubmitTextMass');
            let loader = document.getElementById('submitLoaderMass');
            let btn = document.getElementById('btnSubmitMass');

            let originalText = btnTxt.innerHTML;
            loader.classList.remove('hidden');
            btn.disabled = true;

            let successCount = 0;
            let errorCount = 0;
            let lastUploadNo = "-";
            let totalFreq = "-";

            const BATCH_SIZE = 10;
            let batches = [];
            for (let i = 0; i < massUploadDataValid.length; i += BATCH_SIZE) {
                batches.push(massUploadDataValid.slice(i, i + BATCH_SIZE));
            }

            for (let i = 0; i < batches.length; i++) {
                let currentBatch = batches[i];
                let progressText = `Upload Batch ${i+1}/${batches.length}...`;
                btnTxt.innerHTML = `${progressText}<br><span class="text-[9px]">Memproses ${currentBatch.length} data. Tunggu...</span>`;
                
                try {
                    const response = await fetch(SCRIPT_URL, {
                        method: 'POST', headers: { 'Content-Type': 'text/plain;charset=utf-8' },
                        body: JSON.stringify({ action: 'submitData', userData: currentUserData, data: currentBatch })
                    });
                    
                    const result = await response.json();
                    
                    if (result.status === 'success' || result.status === 'warning') {
                        successCount += (result.summary ? result.summary.successCount : 0);
                        errorCount += (result.summary ? result.summary.duplicateCount : 0);
                        
                        if(result.summary && result.summary.noUpload !== "-") {
                            lastUploadNo = result.summary.noUpload;
                            totalFreq = result.summary.totalHariIni;
                        }
                    } else {
                        errorCount += currentBatch.length;
                    }
                } catch(e) {
                    errorCount += currentBatch.length;
                }
                
                if (i < batches.length - 1) {
                    await new Promise(resolve => setTimeout(resolve, 1000));
                }
            }

            document.getElementById('modNoUpload').innerText = lastUploadNo;
            document.getElementById('modTotalSuccess').innerText = successCount + " Data Masuk";
            document.getElementById('modTotalDuplicate').innerText = errorCount + " Ditolak";
            document.getElementById('modTotal').innerText = totalFreq + "x Upload";

            const hdBg = document.getElementById('modHeaderBg');
            const icBg = document.getElementById('modIconBg');
            const ic = document.getElementById('modIcon');
            const title = document.getElementById('modTitle');
            const sub = document.getElementById('modSubtitle');

            if(successCount > 0) {
                hdBg.className = "bg-green-50 px-6 py-6 border-b border-green-100 text-center";
                icBg.className = "mx-auto flex items-center justify-center h-14 w-14 rounded-full bg-green-100 mb-4";
                ic.className = "fas fa-check text-green-600 text-2xl";
                title.innerText = "Mass Upload Selesai!";
                sub.innerText = "Sistem telah mengecek DataClaim & menyimpan data valid.";
                logActivity(`Mass Upload ${successCount} data sukses`);
            } else {
                hdBg.className = "bg-red-50 px-6 py-6 border-b border-red-100 text-center";
                icBg.className = "mx-auto flex items-center justify-center h-14 w-14 rounded-full bg-red-100 mb-4";
                ic.className = "fas fa-times text-red-600 text-2xl";
                title.innerText = "Upload Ditolak Server!";
                sub.innerText = "Semua baris Excel Anda terdeteksi sudah ada di DataClaim.";
                logActivity(`Mass Upload gagal/ditolak (Duplicate)`);
            }

            document.getElementById('summaryModal').classList.remove('hidden');
            btnTxt.innerHTML = originalText;
            loader.classList.add('hidden');
            resetMassForm();
        }

        async function executeUpload(dataPayload, mode) {
            let btnTxt = document.getElementById('btnSubmitTextSingle');
            let loader = document.getElementById('submitLoaderSingle');
            let btn = document.getElementById('btnSubmitSingle');

            let originalText = btnTxt.innerHTML;
            btnTxt.innerHTML = "Mengupload...";
            loader.classList.remove('hidden');
            btn.disabled = true;

            try {
                const response = await fetch(SCRIPT_URL, {
                    method: 'POST', headers: { 'Content-Type': 'text/plain;charset=utf-8' },
                    body: JSON.stringify({ action: 'submitData', userData: currentUserData, data: dataPayload })
                });

                const result = await response.json();

                if (result.status === 'success' || result.status === 'warning') {
                    showSummaryModal(result.summary, false);
                    logActivity(`Single Upload NoDoc: ${dataPayload.noDoc}`);
                    resetSingleForm();
                } else {
                    showNotification("Error: " + result.message, "error");
                    logActivity(`Gagal Single Upload: ${dataPayload.noDoc}`);
                }
            } catch (error) {
                showNotification("Terjadi kesalahan jaringan/timeout.", "error");
            } finally {
                btnTxt.innerHTML = originalText;
                loader.classList.add('hidden');
                btn.disabled = false;
            }
        }
